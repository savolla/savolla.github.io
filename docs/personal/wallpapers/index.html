Wallpapers

2025-09-13·
Personal -F

These are my favorite wallpapers that I use everyday. Grab whichever you like!

LOCAL_WALLPAPER_DIR="$HOME/resource/images/wallpapers/favorites"
SITE_STATIC_IMG_SUBDIR="$HOME/project/publishing/savolla.github.io/static/images/wallpapers"

mkdir -p "$SITE_STATIC_IMG_SUBDIR"

# Create hardlinks for each image
for file in "$LOCAL_WALLPAPER_DIR"/*; do
    filename=$(basename "$file")
    ln -f "$file" "$SITE_STATIC_IMG_SUBDIR/$filename"
done

# Generate HTML to show the images
echo "#+begin_export html"
for wallpaper in "$SITE_STATIC_IMG_SUBDIR"/*; do
    filename=$(basename "$wallpaper")
    echo "<img style=\"display: inline-block; margin: 5px;\" src=\"/images/wallpapers/$filename\" width=\"190px\" />"
done
echo "#+end_export"
