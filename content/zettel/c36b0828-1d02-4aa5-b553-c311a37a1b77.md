```groovy
pipeline {
  agent none

  environment {
    A = "abc"
  }

  stages {
    stage("conditional stage") {
      steps {
        script {
          if ( A == "abc" ) {
            error "aborting the pipeline"
          }
        }
      }
    }
  }
}
```
